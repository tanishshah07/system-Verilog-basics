     23  task run_phase();

     27  	//-----------------------------------------------------------------
     28  	// 3 write and 3 read;
     29  	   if(count<3) begin 
     30  		   if(!trans.randomize() with {trans.fnx_e==1;}) begin
     31  		    $error("Randomization FAils");

     35  	   end
     36  	   else if(count>=3) begin 
     37  		   if(!trans.randomize() with {trans.fnx_e==2;trans.rd_addr==que[count-3];}) begin
     38  		     $error("Randomization FAils");
     39  		   end
     40                     trans.dis("Sanity generator");
     41  	   end

     44    end
